--- Inline shell script logic (the command block) ---

The script uses /bin/sh -ec (not bash; alpine has no bash). Avoid bash-isms.
All postgres client commands (createdb, dropdb, psql, pg_restore, pg_isready, pg_ctl) connect
to localhost only — never to the Service named "postgres".

Step 0 — install aws-cli:
  apk add --no-cache aws-cli

Step 1 — set local postgres env (NEVER accept these from pod env):
  PGHOST=localhost
  PGPORT=5432
  PGUSER=postgres
  PGPASSWORD is already set from Secret (the scratch postgres will be initialized with this password)
  export AWS_CONFIG_FILE=/scratch/aws-config
  aws configure set default.s3.addressing_style path

Step 2 — refuse-if-live-host safety check (per DRILL-01 defense-in-depth).
  This guard MUST appear VERBATIM as literal shell in the Job command block, BEFORE any
  initdb/createdb/pg_restore step — it is the executable safety barrier, not prose.
  Logic: the scratch drill only ever talks to localhost. If the pod env injects ANY
  POSTGRES_HOST/PGHOST that is not "localhost" (in particular the live Service name
  "postgres"), refuse and exit non-zero. This satisfies must_have truth
  "Script detects PGHOST != localhost at startup and exits 1".

  Exact shell to embed:
  ```sh
  # DRILL-01 safety barrier — refuse if any injected host points off-localhost.
  _injected_host="${POSTGRES_HOST:-${PGHOST:-localhost}}"
  if [ "$_injected_host" != "localhost" ] && [ "$_injected_host" != "127.0.0.1" ]; then
    echo "FATAL: injected DB host '$_injected_host' is not localhost (live Service name is 'postgres') — refusing drill to protect live data (DRILL-01)" >&2
    exit 1
  fi
  # Force every libpq client in this script to localhost regardless of inherited env.
  export PGHOST=localhost PGPORT=5432 PGUSER=postgres
  ```

Step 3 — initialize and start scratch postgres:
  initdb_dir=/scratch/pgdata
  pg_ctl initdb -D "$initdb_dir" --pwfile=<(printf '%s' "$PGPASSWORD") -A md5
  Note: alpine sh has no process substitution (<(...)). Use a temp file instead:
    printf '%s' "$PGPASSWORD" > /scratch/pgpwfile
    pg_ctl initdb -D "$initdb_dir" -A md5 --pwfile=/scratch/pgpwfile
    rm /scratch/pgpwfile
  Then start:
    pg_ctl -D "$initdb_dir" -l /scratch/postgres.log start
  Wait for ready (loop 30 iterations with sleep 1, using pg_isready -h localhost -U postgres):
    i=0; while [ $i -lt 30 ]; do
      if pg_isready -h localhost -p 5432 -U postgres >/dev/null 2>&1; then
        echo "scratch postgres ready"; break; fi
      i=$((i+1)); sleep 1
    done
    if ! pg_isready -h localhost -p 5432 -U postgres >/dev/null 2>&1; then
      echo "ERROR: scratch postgres did not start"; exit 1; fi

Step 4 — discover latest backup (per CONTEXT.md lexicographic max decision):
  backup_id=$(aws --endpoint-url="$S3_ENDPOINT" \
    s3 ls "s3://${S3_BUCKET}/${BACKUP_PREFIX}/" \
    | awk '{print $NF}' | sed 's|/$||' | sort -r | head -1)
  Verify backup_id is non-empty and that manifest.json exists for it:
    if [ -z "$backup_id" ]; then echo "ERROR: no backups found"; exit 1; fi
    if ! aws --endpoint-url="$S3_ENDPOINT" s3 ls \
        "s3://${S3_BUCKET}/${BACKUP_PREFIX}/${backup_id}/manifest.json" >/dev/null 2>&1; then
      echo "ERROR: manifest.json missing for backup_id=${backup_id}"; exit 1; fi
    echo "backup_id=${backup_id}"

Step 5 — download dump and list files:
  aws --endpoint-url="$S3_ENDPOINT" \
    s3 cp "s3://${S3_BUCKET}/${BACKUP_PREFIX}/${backup_id}/solid_stats.dump" \
    /scratch/solid_stats.dump --only-show-errors
  aws --endpoint-url="$S3_ENDPOINT" \
    s3 cp "s3://${S3_BUCKET}/${BACKUP_PREFIX}/${backup_id}/solid_stats.dump.list" \
    /scratch/solid_stats.dump.list --only-show-errors

Step 6 — create guarded scratch database and restore:
  createdb -h localhost -U postgres solid_stats_drill
  pg_restore --host=localhost --username=postgres \
    --dbname=solid_stats_drill --no-owner --no-privileges \
    /scratch/solid_stats.dump || true
  (use || true because pg_restore exits non-zero on harmless warnings like "already exists")

Step 7 — sanity assertions with captured result code (Pattern 3 from RESEARCH.md):
  drill_result=0
  drill_start=$(date +%s)

  Assertion A — table count >= 5:
    table_count=$(psql -h localhost -U postgres -d solid_stats_drill -t -A \
      -c "SELECT COUNT(*) FROM information_schema.tables WHERE table_schema = 'public';")
    if [ "$table_count" -lt 5 ]; then
      echo "FAIL: table_count=${table_count} (expected >= 5)"; drill_result=1; fi

  Assertion B — at least one non-empty table (check information_schema row count as proxy):
    row_count=$(psql -h localhost -U postgres -d solid_stats_drill -t -A \
      -c "SELECT SUM(n_live_tup) FROM pg_stat_user_tables;")
    if [ "${row_count:-0}" -lt 1 ]; then
      echo "FAIL: total_rows=${row_count} (expected > 0)"; drill_result=1; fi

  Assertion C — dump list file is non-empty:
    list_lines=$(wc -l < /scratch/solid_stats.dump.list)
    if [ "$list_lines" -lt 1 ]; then
      echo "FAIL: dump list is empty"; drill_result=1; fi

  Emit result line (structured evidence for DRILL-03):
    drill_end=$(date +%s)
    drill_duration=$((drill_end - drill_start))
    if [ $drill_result -eq 0 ]; then
      echo "DRILL_RESULT=PASS backup_id=${backup_id} table_count=${table_count} total_rows=${row_count} duration_s=${drill_duration}"
    else
      echo "DRILL_RESULT=FAIL backup_id=${backup_id} table_count=${table_count} total_rows=${row_count} duration_s=${drill_duration}"
    fi

Step 8 — teardown (always runs, never masks result):
  dropdb -h localhost -U postgres solid_stats_drill || true
  pg_ctl -D "$initdb_dir" stop || true
  rm -rf /scratch/pgdata /scratch/solid_stats.dump /scratch/solid_stats.dump.list

Step 9 — exit with captured result:
  exit $drill_result

Script behavior:

   namespace="${K8S_NAMESPACE:-solid-stats-staging}"
   timeout="${DRILL_TIMEOUT:-900s}"

3. Delete any pre-existing restore-drill jobs with the same name (idempotency):
   kubectl -n "$namespace" delete job "$job_name" --ignore-not-found

4. Apply manifest with name override via --field-manager-style patch using kubectl create:
   kubectl -n "$namespace" create job "$job_name" is NOT available from a manifest file.
   Instead, use the standard approach: apply the manifest then patch the name.
   Since Job names in the manifest are fixed as "restore-drill", we use a unique name by:
     kubectl -n "$namespace" apply -f "$manifest"
   This creates/updates the Job named "restore-drill" in the manifest.
   To support multiple concurrent runs the operator should ensure no prior job "restore-drill"

   if ! kubectl -n "$namespace" wait \
       --for=condition=complete "job/$job_name" --timeout="$timeout" 2>/dev/null; then
     # Job may have failed (condition=failed) rather than timed out
     echo "Drill Job did not complete — checking for failure..." >&2
     kubectl -n "$namespace" describe "job/$job_name" || true
     kubectl -n "$namespace" logs "job/$job_name" --all-containers=true || true
     exit 1
   fi

   logs="$(kubectl -n "$namespace" logs "job/$job_name" --all-containers=true)"
   printf '%s\n' "$logs"

7. Extract and validate DRILL_RESULT line:
   result_line="$(printf '%s\n' "$logs" | grep '^DRILL_RESULT=' | tail -1)"
   if [ -z "$result_line" ]; then
     echo "error: no DRILL_RESULT line found in job logs" >&2; exit 1; fi

   echo ""
   echo "=== Restore Drill Evidence ==="
   echo "$result_line"

   if printf '%s\n' "$result_line" | grep -q 'DRILL_RESULT=FAIL'; then
     echo "RESTORE DRILL FAILED" >&2; exit 1; fi

   echo "RESTORE DRILL PASSED"

8. Cleanup Job (pod self-cleans via ttlSecondsAfterFinished; we clean the Job object now):
   kubectl -n "$namespace" delete "job/$job_name" --ignore-not-found
